## 题目

```
统计所有小于非负整数 n 的质数的数量。
```

## 解法

### 代码

```c++
/* 暴力法 */
class Solution {
public:
    int countPrimes(int n) {
        int cnt = 0;
        for (int i = 1; i < n; ++i) {
            if (isPrime(i)) cnt++;
        }
        return cnt;
    }

    bool isPrime(int x) {
        if (x <= 1) return false;
        if (x == 2) return true;
        for (int i = 2; i < x; ++i) {
            if (x % i == 0) return false;
        }
        return true;
    }
};
```

```c++
/* 六倍法 */
class Solution {
public:
    int countPrimes(int n) {
        int cnt = 0;
        for (int i = 1; i < n; ++i) {
            if (isPrime(i)) cnt++;
        }
        return cnt;
    }

    bool isPrime(int x) {
        if (x == 2 || x == 3) return 1;
        if (x == 1 || x % 6 != 1 && x % 6 != 5) return 0;

        for (int i = 5; i <= sqrt(x); i += 6) {
            if (x % i == 0 || x % (i + 2) == 0) return false;
        }
        return true;
    }
};
```

```c++
/* Miller-Rabin 素数判定，在 Fermat 基础上增加了二次判定 */
class Solution {
public:
    int countPrimes(int n) {
        int cnt = 0;
        for (int i = 1; i < n; ++i) {
            if (isPrime(i)) cnt++;
        }
        return cnt;
    }

    // rand 2 - e
    int rand_Number(int e) {
        return rand() % e + 1;;
    }

    // a * b % c
    int mult_mod(int a, int b, int c) {
        a %= c;
        b %= c;
        int res = 0;
        while (b) {
            if (b & 1) {
                res += a;
                res %= c;
            }
            a <<= 1;
            if (a >= c)
                a %= c;
            b >>= 1;
        }
        return res;
    }

    // a^u % num
    int getRemainder(int a, int u, int num) {
        int cur = 1;
        int nxt = a;
        while (u) {
            if ((u & 1) > 0) cur = mult_mod(cur, nxt, num);
            nxt = mult_mod(nxt, nxt, num);
            u = u >> 1;
        }
        return cur % num;
    }

    bool isPrime(int num) {
        if (num == 2) return true;
        if (num < 2 || num % 2 == 0) return false;
        int u = num - 1;
        int S = 20; // 检测次数
        while (u % 2 == 0) {
            u /= 2;
        }
        for (int i = 0; i < S; i++) {
            int a = rand_Number(num - 1);
            int x = getRemainder(a, u, num);
            int y = x;
            int tu = u;
            while (tu < num) {
                y = mult_mod(x, x, num);;
                if (y == 1 && x != 1 && x != num - 1) return false;
                x = y;
                tu *= 2;
            }
            if (x != 1) return false;
        }
        return true;
    }
};
```
